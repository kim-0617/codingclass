js

이번엔 requestAnimationFrame이라는 함수를 사용해봤는데요? scroll 함수 안에서 계속 재귀적으로 호출 해주는데, 앞서 언급한 requestAnimationFrame()의 인자로 scroll 함수를 넘겨주면 브라우저가 뻗지않고 효율적으로 스크롤을 감지한다고 하네요. (1초에 60번이라고 했던가요? 호출횟수가)

코드 보기 
function scroll() {
    let scrollTop = window.pageYOffset || window.scrollY || document.documentElement.scrollTop;
    document.querySelectorAll('.content__item').forEach(item => {
        if(scrollTop > item.offsetTop - window.innerHeight / 2) {
            item.classList.add('show');
        }
    });
    requestAnimationFrame(scroll);
}
