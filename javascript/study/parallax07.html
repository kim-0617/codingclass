css

reveal 클래스가 붙지 않았을 때는 opacity를 0으로 하고 show 클래스가 붙었을 때는, opacity 애니메이션을 통해 해당 영역을 드러내줍니다. 다양한 클래스를 생성해서, reveal-BTT(bottom to top) reveal-RTL(Right to left) 등을 선언해서 다양한 방향에서 나타나는 효과를 줄 수 있도록 설정했습니다.

js

자바스크립트의 역할은 show를 붙여주는데 쓰일 뿐 입니다.

코드 보기 
// reveal 클래스 자식에 글씨만 있다면 가상으로 span 태그 만들어서 넣어주세요
const reveal = document.querySelectorAll('.reveal');
reveal.forEach((el) => {
    let text = el.innerText;
    if (text) {
        el.innerHTML = "" + text + "";
    }
});

function scroll() {
    let scrollTop = window.scrollY;
    reveal.forEach((el, index) => {
        let revealOffset = el.offsetTop + el.parentElement.offsetTop;
        let revealDelay = el.dataset.delay;

        // if(scrollTop > revealOffset - window.innerHeight) {
        //     el.classList.add('show');
        // }

        if (scrollTop > revealOffset - window.innerHeight / 2) {
            if (!revealDelay) {
                el.classList.add('show');
            }
            else {
                setTimeout(() => {
                    el.classList.add("show");
                }, revealDelay);
            }
        }
    });
    requestAnimationFrame(scroll);
}
