조건문

조건문은 조건에 참 거짓에 따라 동작을 다르게 수행하고 싶을 때 사용하는 문법입니다.

if (조건문) {
    조건이 참일 때 실행 될 문장
}
else if(조건문) {
    if 문의 조건에 맞지 않고, 조건이 참일 때 실행 될 문장
}
else{
    위의 모든조건에 해당되지 않을 때 실행될 문장
}

조건에 따라 입장료 알려주기

나이에 따라 지불해야 할 입장료를 알려주는 코드

var age = prompt("당신의 나이는?");
var addmissionFee = '';
if(age < 13 || age >= 65){
    addmissionFee = '2,000원';
}
else if(age >= 13 && age < 20){
    addmissionFee = '4000원';
}
else{
    addmissionFee = '6000원';
}
document.write(addmissionFee);
결과 확인하기
20을 입력했을 때,
6000원

조건에 따라 본인의 합격 여부 알려주기

다양한 조건의 맞춰 자신의 점수가 합격인지 불합격인지 알려주는 코드

var html = Number(prompt('html 점수 입력', ''));
var css = Number(prompt('css 점수 입력', ''));
var javascript = Number(prompt('javascript 점수 입력', ''));
var avg = (html + css + javascript) / 3; // 평균
var comment;
if(avg >= 60){
    if(html > 40){
        comment = 'HTML' + ' 40점 미만 불합격';
    }
    else if(css > 40){
        comment = 'CSS' + ' 40점 미만 불합격';
    }
    else if(javascript > 40){
        comment = 'JAVASCRIPT' + ' 40점 미만 불합격';
    }
    else{
        comment = '평균 ' + avg + '점 합격';
    }
}
else{
    comment = '평균' + avg + '점 불합격';
}
document.write(comment);
결과 확인하기
60 60 60 일 경우,
평균 60점 합격

switch문을 사용하여 입장료 판단하기

아동, 청소년, 성인을 입력받아서, 입장료를 알려주는 코드

var classification = prompt('아동, 청소년, 성인 중 선택 입력', '');
switch(classification){
    case '아동':
        classification += ' : 입장료 무료';
        break;
    case '청소년':
        classification += ' : 입장료 2,000원';
        break;
    case '성인':
        classification += ' : 입장료 5,000원';
        break;
    default:
        classification += '입력 값을 다시 확인하세요';
}
document.write(classification);
결과 확인하기
청소년 입력 시,
청소년 : 입장료 2,000원
