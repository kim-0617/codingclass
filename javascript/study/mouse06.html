js

마우스 커서는 계속 해왔던 내용 입니다. .spanSlow에는 좌표를 센터값으로 초기화한 (e.pageX - window.innerWidth / 2) 이 값 있죠? 지난 마우스 이펙트부터 계속 사용해 왔던 방식이요.. 그 수치에 slow에는 0.1, fast에는 0.2 곱해주고 그 값을 gsap로 x 값을 움직이면 마우스를 움직일 때 마다 부드럽게 글씨가 움직이겠죠 

const mouseMove = (e) => {
    let positionSlow = (e.pageX - window.innerWidth / 2) * 0.1 ;
    let positionFast = (e.pageX - window.innerWidth / 2) * 0.2 ;

    gsap.to(".spanSlow", {
        duration : .4,
        x : positionSlow,
    });

    gsap.to(".spanFast", {
        duration : .4,
        x : -positionFast,
    });

    gsap.to(".mouse__cursor", {
        duration : .3,
        left : e.pageX,
        top : e.pageY,
        xPercent : -50,
        yPercent : -50,
    });
}
window.addEventListener('mousemove', mouseMove);

// 마우스 커서가 글씨 위에 올라가면 크기는 5배 정도 크게, 글씨는 반전효과
const spans = document.querySelectorAll('.mouse__text span');
const cursor = document.querySelector('.mouse__cursor');
cursor.style.transition = "scale 1s";
spans.forEach((span, idx) => {
    span.addEventListener('mouseenter', (e) => {
        cursor.style.scale = "10";
    });

    span.addEventListener('mouseleave', (e) => {
        cursor.style.scale = "1";
    });
});
