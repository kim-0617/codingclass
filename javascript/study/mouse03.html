js

자 이제 js로 커서를 움직일 건데, gsap로 움직이는건 2번해서 해봤죠? 크게 설명할건 없을 것 같습니다. 

const cursor = document.querySelector('.mouse__cursor');
const circleWidth = cursor.offsetWidth || cursor.clientWidth // 보더포함 vs 보더제외
const circleHeight = cursor.offsetHeight || cursor.clientHeight // 보더포함 vs 보더제외

const circleInfo = cursor.getBoundingClientRect();

window.addEventListener('mousemove', function(e) {
    gsap.to(cursor, {
        duration : 0.3,
        left : e.pageX - circleInfo.width / 2,
        top : e.pageY - circleInfo.height / 2,
        // xPercent : -50,
        // yPercent : -50,
    });
});
