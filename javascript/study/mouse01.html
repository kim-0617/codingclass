js

맨 위에 부분 mousemove 이벤트는 좌표 값 보여주는 코드입니다. (1번째 부분)
그 밑에 코드가 마우스가 움직일 때 마다, cursor부분의 position 값이 실시간으로 변경됩니다. (2번째 부분)
그 후, 강조 표시한 단어에 hover 했을 때 마다 효과를 다르게 주기 위해서 span 태그에 스타일을 넣어서 css에 정의 해놓고, (3번째 부분)
정의해 놓은 영역에 mouseover 했을 때 cursor에 class를 추가시켜줘서 style에 맞는 class를 추가 시켜줍니다. 

// 마우스 좌표 구하기
window.addEventListener("mousemove", (event) => {
    document.querySelector(".clientX").innerText = event.clientX;
    document.querySelector(".clientY").innerText = event.clientY;
    document.querySelector(".offsetX").innerText = event.offsetX;
    document.querySelector(".offsetY").innerText = event.offsetY;
    document.querySelector(".pageX").innerText = event.pageX;
    document.querySelector(".pageY").innerText = event.pageY;
    document.querySelector(".screenX").innerText = event.screenX;
    document.querySelector(".screenY").innerText = event.screenY;
});

// 커서 포인터 따라다니게 하기
window.addEventListener("mousemove", (e) => {
    const cursor = document.querySelector('.mouse__cursor');
    cursor.style.left = e.clientX - 25 + "px";
    cursor.style.top = e.clientY - 25 + "px";
});

document.querySelectorAll('.mouse__wrap span').forEach((span) => {
    const spanClass = span.getAttribute('class');
    span.addEventListener("mouseover", () => {
        const cursor = document.querySelector('.mouse__cursor');
        cursor.classList.add(spanClass);
    });
    span.addEventListener("mouseout", () => {
        const cursor = document.querySelector('.mouse__cursor');
        cursor.classList.remove(spanClass);
    });
});
