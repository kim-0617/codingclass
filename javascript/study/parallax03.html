js

자 js 부분인데요? 순서대로 가보겠습니다.
1. window.addEventListener('scroll', scrollProgress) 해주면 setInterval이 마우스를 스크롤 할 때 마다 호출 되겠죠?
2. 스크롤을 움직이면 nowScroll = true가 실행되면서 if문이 실행됩니다. 이게 없으면 false로 해놓은걸 돌릴수가없어서 hasScroll이 실행되지 못하겠죠?
3. 트리거 변수가 없다면 하루종일 scrollTop === lastScroll 이 되서 메뉴는 나오지도 않습니다.
4. 저기 nowScroll = false; 이게 없어도 마찬가지입니다.
5. 자 이제 hasScroll 부분인데요? 그전에 hasScroll 함수의 실행됐던 scrollTop 값이 lastScroll 값에 저장 되있겠죠?
6. 자 이제 스크롤을 위로 하는 순간 hasScroll이 실행되면서 lastScroll(전에 봤던 값) > scrollTop(올렸다면 더 작겠죠? 이 값이)
7. 그러면 어떻게 되요? show가 붙으니까 메뉴가 나오겠죠? 끝입니다.

let nowScroll = true;
let lastScroll = 0;

function scrollProgress() {
    let scrollTop = window.pageYOffset || window.scrollY || document.documentElement.scrollTop;
    document.querySelector('#parallax__info span').innerText = scrollTop;

    if(scrollTop > (document.documentElement.scrollHeight - window.innerHeight) - 200){
        document.querySelector('#parallax__top').classList.add('show');
    }
    else{
        document.querySelector('#parallax__top').classList.remove('show');
    }
    nowScroll = true;
    setInterval(() => {
        console.log("실행 : ", nowScroll)
        if(nowScroll){
            hasScroll();
            nowScroll = false;
        }
    }, 1000);
}

function hasScroll() {
    let scrollTop = window.pageYOffset || window.scrollY || document.documentElement.scrollTop;
    console.log("현재 값",scrollTop, "이전 값", lastScroll)
    if(scrollTop < lastScroll){
        document.querySelector('#parallax__nav').classList.add('show');
    }
    else{
        document.querySelector('#parallax__nav').classList.remove('show');
    }
    lastScroll = scrollTop;
}

window.addEventListener('scroll', scrollProgress);
