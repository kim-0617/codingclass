js

이번에도 역시 requestAnimationFrame을 사용해서 scroll 함수를 재귀호출 해주었습니다. 이미지와 설명에(명언부분) transform 효과를 줄 건데, scrollTop - 섹션의 offsetTop 값은 너무 크기 때문에 0.1을 곱해서 적당량 만큼 애니메이션 효과가 들어가도록 해줬습니다. 그런데 그냥 transform 효과는 부드럽지 못하기 때문에 gsap를 사용하여 효과를 주었습니다.

코드 보기 
window.addEventListener('scroll', () => {
    let scrollTop = window.pageYOffset || window.scrollY || document.documentElement.scrollTop;
    document.querySelector('#parallax__info span').innerText = scrollTop;
});
function scroll() {
    let scrollTop = window.pageYOffset || window.scrollY || document.documentElement.scrollTop;
    document.querySelectorAll('.content__item').forEach((item) => {
        const target1 = item.querySelector('.content__item__img');
        const target2 = item.querySelector('.content__item__desc');
        let offset = (scrollTop - item.offsetTop) * 0.1;
        let offset2 = (scrollTop - item.offsetTop) * 0.1;
        // target1.style.transform = `translateY(${offset}px)`;
        // target2.style.transform = `translateX(${offset2}px)`;

        gsap.to(target1, {
            duration : .1,
            y : offset,
            ease : "power4.out",
        });
        gsap.to(target2, {
            duration : .3,
            y : offset2,
        });
    });
    requestAnimationFrame(scroll);
}
scroll();
