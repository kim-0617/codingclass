js

charA 메서드를 이용해서, 만들어둔 알파벳 버튼을 클릭하면 해당 알파벳으로 시작하는 속성들만 보여주도록 했습니다. 해당 알파벳과 모든 css속성들의 첫글자가 같으면 hide 클래스를 제거해서 볼 수 있도록 했어요! 

// 선택자
const searchKeyword = document.querySelectorAll('.search__info .keyword span'); // 알파벳
const searchList = document.querySelectorAll('.search__list ul li'); // 목록 리스트
const searchInfo = document.querySelector('.search__info .num'); // 목록 리스트

// 속성 갯수 설정하기
searchInfo.innerHTML = searchList.length;

//
searchKeyword.forEach((el) => {
el.addEventListener('click', () => {
    searchKeyword.forEach((del) => {
        del.classList.remove('active');
    });
    let cnt = 0;
    const searchWord = el.innerText; // 사용자가 클릭한 알파벳.
    el.classList.add('active');

    searchList.forEach(el => {
        const cssName = el.querySelector('strong').innerText;

        // 알파벳 첫글자 === css 속성의 첫글자
        if (searchWord.charAt(0) === cssName.charAt(0)) {
            el.classList.remove('hide');
        }
        else {
            el.classList.add('hide');
        }

        if (!el.classList.value) {
            cnt++;
        }

        searchInfo.innerHTML = cnt;
    });
});
});
