이번엔 더 간단하죠? scrollTop 값 정보를 똑같이 적어줄게요. 메뉴를 클릭했을 때, 기본동작을 막아주고 스크롤 이동할 때 스무스 효과를 줄게요. 그리고 저번에 했던거 있죠? scrollTop값과 offsetTop값 비교해서 해당 섹션에 위치하면 active 효과를 줘야겠죠?

코드 보기 
document.querySelectorAll('#parallax__dot a').forEach((el) => {
    el.addEventListener("click", e => {
        e.preventDefault();

        document.querySelector(el.getAttribute('href')).scrollIntoView({behavior : "smooth",});
    });
});

window.addEventListener('scroll', () => {
    let scrollTop = window.pageYOffset || window.scrollY || document.documentElement.scrollTop;
    document.querySelector('#parallax__info span').innerText = scrollTop;
    document.querySelectorAll('.content__item').forEach((el, index) => {
        if(scrollTop >= el.offsetTop - window.innerHeight/2) {
            document.querySelectorAll('#parallax__dot li').forEach((li) => {
                li.classList.remove('active');
            });

            document.querySelector(`#parallax__dot li:nth-child(${index + 1})`).classList.add('active');
        }
    });
});
