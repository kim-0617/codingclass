버튼을 누르면 해당 텍스트에 맞게 정렬하는 부분입니다. 여기서 크게 언급드릴게 없으니 이만 줄이겠습니다. 

const cssList = document.querySelector('.search__list .line'); // css 목록들이 들어갈 자리

const listCssArr = [];
// 데이터 출력하기
const updateList = () => {
    let listCss = "";
    cssProperty.forEach((data, index) => {
        listCss += `<li>${data.num}. ${data.name} : ${data.desc}</li>`;
        listCssArr.push(`<li>${data.num}. ${data.name} : ${data.desc}</li>`);
    });
    cssList.innerHTML = listCss;
}
updateList();

// 정렬 함수 => 앞에 것이 클 때 오름차순 정렬
function sortReverse() {
    cssProperty.reverse();
    updateList();
}

function sortDesc() {
    cssProperty.sort((a,b) => b.num - a.num);
    updateList();
}

function sortAsc() {
    cssProperty.sort((a,b) => a.num - b.num);
    updateList();
}

function sortAlphabet() {
    cssProperty.sort((prev, cur) => { 
        const [upperPrev, upperCur] = [prev.name.toUpperCase(), cur.name.toUpperCase()]
        if (upperPrev > upperCur) return 1;
        if (upperPrev < upperCur) return -1;
        // if (upperPrev === upperCur) return 0;
    });
    updateList();
}

function sortAlphabetZ() {
    cssProperty.sort((prev, cur) => { 
        const [upperPrev, upperCur] = [prev.name.toUpperCase(), cur.name.toUpperCase()]
        if (upperPrev < upperCur) return 1;
        if (upperPrev > upperCur) return -1;
        // if (upperPrev === upperCur) return 0;
    });
    updateList();
}

// 반대로 정렬 클릭
const reverseBtn = document.querySelector('.btn1');
reverseBtn.addEventListener('click', sortReverse);

// 내림차순 정렬 클릭
const descBtn = document.querySelector('.btn2');
descBtn.addEventListener('click', sortDesc);

// 오름차순 정렬 클릭
const ascBtn = document.querySelector('.btn3');
ascBtn.addEventListener('click', sortAsc);

// 알파벳 순으로 정렬
const alphaBtn = document.querySelector('.btn4');
alphaBtn.addEventListener('click', sortAlphabet);

// 알파벳 역순으로 정렬
const alphaReverseBtn = document.querySelector('.btn5');
alphaReverseBtn.addEventListener('click', sortAlphabetZ);

// 피셔 에이츠 (랜덤 정렬)
function shuffle(data) {
    for(let i = data.length -1; i >= 0; i--) {
        const randomIndex = Math.floor(Math.random() * (i + 1));
        [data[i], data[randomIndex]] = [data[randomIndex], data[i]];
        cssList.innerHTML += data[i];
    }
}

const randomBtn = document.querySelector('.btn6');
randomBtn.addEventListener('click', (e) => {
    e.preventDefault();
    cssList.innerHTML = '';
    shuffle(listCssArr);
});
