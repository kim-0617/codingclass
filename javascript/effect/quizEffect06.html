const quizType = document.querySelector('span.quiz__type'); // 문제 번호, 질문
const quizQuestion = document.querySelector('.quiz__question'); // 문제 번호, 질문
const quizSelects = document.querySelector('.quiz__selects'); // 보기
const quizResult = document.querySelector('.quiz__answer .result'); // 해설
const quizConfirm = document.querySelector('.quiz__answer .confirm'); // 정답 확인 버튼
const quizView = document.querySelector('.quiz__view')
let quizCount = 0;
let quizScore = 0;
let clickable = true;
let eraser = [];
let inputTags = [];
// 문제 출력
const updateQuiz = (index) => {
    eraser = [];
    inputTags = [];
    if(index >= note.length) return;
    const questionTag = `
    <span class="number">${note[index].answerNum}. </span>
    <div class="ask">${note[index].answerAsk}</div>
    `;
    const choiceTag = `
    <label for="select1">
        <input type="radio" name="select" value="1" id="select1" class="select">
        <span class="choice">${note[index].answerChoice[0]}</span>
    </label>
    <label for="select2">
        <input type="radio" name="select" value="2" id="select2" class="select">
        <span class="choice">${note[index].answerChoice[1]}</span>
    </label>
    <label for="select3">
        <input type="radio" name="select" value="3" id="select3" class="select">
        <span class="choice">${note[index].answerChoice[2]}</span>
    </label>
    <label for="select4">
        <input type="radio" name="select" value="4" id="select4" class="select">
        <span class="choice">${note[index].answerChoice[3]}</span>
    </label>
    `;

    // 문제 출력
    quizType.innerHTML = note[index].answerType;
    quizQuestion.innerHTML = questionTag;
    quizSelects.innerHTML = choiceTag;
    quizResult.innerHTML = note[index].answerEx;

    const quizChoice = quizSelects.querySelectorAll('.choice'); // span
    const quizInput = quizSelects.querySelectorAll('.select'); // input
    for (let i = 0; i < quizChoice.length; i++) {
        quizChoice[i].setAttribute("onclick", "choiceSelected(this)");
        eraser.push(quizChoice[i]);
        inputTags.push(quizInput[i]);
    }

    // 문제, 해설 숨기기
    quizResult.style.display = 'none';
    quizConfirm.style.display = 'none';
}
updateQuiz(quizCount);

const choiceSelected = (answer) => {
    let userAnswer = answer.textContent; // 사용자가 체크한 정답
    let currentAnswer = note[quizCount].answerResult;
    let flag = 0;
    if (userAnswer === currentAnswer) {
        quizView.classList.add('like');
        quizScore++;
    }
    else {
        quizView.classList.add('dislike');
        quizResult.style.display = 'block';
    }
    quizConfirm.style.display = 'block';
    eraser.forEach((span,idx) => {
        if(answer === span) flag = idx;
        span.setAttribute('onclick', '');
    });
    inputTags.forEach((input, idx) => {
        if(flag !== idx) input.disabled = true;
    })
}

// 정답 확인 버튼
const answerQuiz = () => {
    // 마지막 문제
    if (note.length - 1 === quizCount) {
        quizConfirm.textContent = `총 ${quizCount + 1}문제 중에 ${quizScore}문제를 맞추었습니다.`
    }
    quizCount++;
    updateQuiz(quizCount);
    quizView.classList.remove('like', 'dislike');
}
quizConfirm.addEventListener('click', answerQuiz);
